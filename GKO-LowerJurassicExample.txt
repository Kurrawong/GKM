Lower_Jurassic_Epoch
	rdf:type Epoch;
	rdfs:comment 'Epoch is a subclass of Geochronologic Era';
	rdfs:label  'Lower Jurassic time interval';
	rdfs:comment 'This is a temporal interval';
	isPartOf  <http://resource.geosciml.org/vocabulary/timescale/isc2017>;
	hasYoungerDate 
		[rdfs:type Geochronologic Boundary;
		hasDate 	174.1 ;
		hasUncertainty 1.0;
		hasUOM [rdfs:type uom:MillionYearsBP];
		isRealizedBy <http://resource.geosciml.org/classifier/ics/ischart/BaseMiddleJurassic>;
		rdfs:comment 'the Base Middle Jurassic class represents all rock body surfaces correlated with the surface at the Global Stratotype Point';
		stratotype <http://resource.geosciml.org/classifier/ics/ischart/BaseMiddleJurassicSP>
		];
	hasOlderDate 
		[rdfs:type Geochronologic Boundary;
		hasDate 201.3 ;
		hasUncertainty 0.2;
		hasUOM [rdfs:type uom:MillionYearsBP];
		isRealizedBy  <http://resource.geosciml.org/classifier/ics/ischart/BaseJurassic>;
		stratotype <http://resource.geosciml.org/classifier/ics/ischart/BaseJurassicSP>
		].
